##  将整数强制拆分为 string
```cpp
#include<cstdio>
#include<cstring>
#include<string>
using namespace std;
int main(){
	long long v7=28537194573619560;
	char *p=(char *)&v7;//chr指针
	char str[30];
	strcpy(str,":\"AL_RT^L*.?+6/46");
	for(int i=0;i<19;i++){
		str[i]=p[i%7]^str[i];
		printf("%c",str[i]);
	}
	return 0;
}
```
