# 「一本通 5.1 例 2」能量项链
`一本通` `区间DP` 
- 同石子合并
```cpp
#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
const int maxn=210;
int n,f[maxn][maxn];
int ll[maxn];
int main()
{
	scanf("%d",&n);
	for(int i=1;i<=n;i++)
	{
		scanf("%d",&ll[i]);
		ll[i+n]=ll[i];
	}
	for(int l=1;l<n*2;l++)
	for(int i=1;i+l<=n*2;i++)
	{
		int j=i+l;
		for(int k=i;k<j;k++)
		f[i][j]=max(f[i][j],f[i][k]+f[k+1][j]+ll[i]*ll[k+1]*ll[j+1]);
	}
	int ans=0;
	for(int i=1;i<=n;i++)ans=max(ans,f[i][i+n-1]);
	printf("%d\n",ans);
	return 0;
}
```
