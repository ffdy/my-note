# 「一本通 3.4 练习 1」糖果
`一本通` `差分约束` `最短路` `环`
- 差分裸题
```cpp
#include<cstdio>
#include<queue>
#include<cstring>
#include<algorithm>
#define ll long long
using namespace std;
const int maxn=1e5+10;
int n,m,head[maxn],num;
ll dis[maxn];
bool use[maxn],fl,vis[maxn];
struct fy{int to,d,next;}q[maxn<<1];
void add(int a,int b,int c){q[++num]=(fy){b,c,head[a]};head[a]=num;}
void sp(int a)
{
	use[a]=true;vis[a]=true;
	for(int i=head[a];i;i=q[i].next)
	{
		int b=q[i].to;
		if(dis[b]<dis[a]+q[i].d)
		{
			dis[b]=dis[a]+q[i].d;
			if(use[b]){fl=true;return;}
			sp(b);
			if(fl)return;
		}
	}
	use[a]=false;
}
int main()
{
	scanf("%d%d",&n,&m);int a,b,c;
	for(int i=1;i<=m;i++)
	{
		scanf("%d%d%d",&c,&a,&b);
		if(c==1){add(a,b,0);add(b,a,0);}
		if(c==2){add(a,b,1);}
		if(c==3){add(b,a,0);}
		if(c==4){add(b,a,1);}
		if(c==5){add(a,b,0);}
		if(a==b&&!c%2)
		{
			printf("-1\n");
			return 0;
		}
	}
	for(int i=1;i<=n;i++)if(!vis[i])
	{dis[i]=1;sp(i);}
	ll ans=0;
	for(int i=1;i<=n;i++)ans+=dis[i];
	if(fl)printf("-1\n");
	else printf("%lld\n",ans);
	return 0;
}
```
