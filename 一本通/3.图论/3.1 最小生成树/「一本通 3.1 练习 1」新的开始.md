# 「一本通 3.1 练习 1」新的开始
`一本通` `生成树`

- 树外加点

```cpp
#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
const int maxn=310;
int n,v[maxn],head[maxn],num,fa[maxn];
struct fy
{
	int from,to,d;
	bool operator<(const fy &a)const{return d<a.d;};
}q[maxn*maxn<<1];
int find(int a)
{
	while(fa[a]!=a)a=fa[a]=fa[fa[a]];
	return a;
}
int main()
{
	scanf("%d",&n);
	for(int i=1;i<=n;i++)
	{
		scanf("%d",&v[i]);
		q[++num]=(fy){0,i,v[i]};
		fa[i]=i;
	}
	for(int i=1;i<=n;i++)for(int j=1;j<=n;j++)
	{
		scanf("%d",&q[++num].d);
		q[num].from=i;q[num].to=j;
	}
	sort(q+1,q+1+num);int ans=0,e=0;
	for(int i=1;i<=num;i++)
	{
		int a=find(q[i].from);
		int b=find(q[i].to);
		if(a!=b)
		{
			e++;
			ans+=q[i].d;
			fa[a]=b;
			if(e==n)
			{
				printf("%d\n",ans);
				return 0;
			}
		}
	}
	return 0;
}
```
