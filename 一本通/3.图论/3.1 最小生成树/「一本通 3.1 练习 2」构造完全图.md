# 「一本通 3.1 练习 2」构造完全图
`一本通` `生成树`

- 使两点连通的边的权值+1

```cpp
#include<cstdio>
#include<cstring>
#include<algorithm>
#define ll long long
using namespace std;
const int maxn=1e5+10;
int n,fa[maxn],size[maxn];
ll ans;
struct fy{int from,to,d;bool operator<(const fy &a)const{return d<a.d;};}q[maxn];
int find(int a)
{
	if(fa[a]!=a)
	{
		size[fa[a]]+=size[a];
		size[a]=0;
		fa[a]=find(fa[a]);
	}
	return fa[a];
}
int main()
{
	scanf("%d",&n);int a,b,c;
	for(int i=1;i<=n;i++){fa[i]=i;size[i]=1;}
	for(int i=1;i<n;i++)scanf("%d%d%d",&q[i].from,&q[i].to,&q[i].d);
	sort(q+1,q+n);
	for(int i=1;i<n;i++)
	{
		int a=find(q[i].from);
		int b=find(q[i].to);
		ans+=q[i].d;
		ans+=(ll)(q[i].d+1)*(size[a]*size[b]-1);
		fa[a]=b;
		size[b]+=size[a];
		size[a]=0;
//		for(int j=1;j<n;j++)printf("%d ",size[j]);
//		printf("\n");
	}
	printf("%lld\n",ans);
	return 0;
}
```
