# 「一本通 1.1 练习 1」数列极差

`一本通` `贪心`

- 贪心每次取极值

```cpp
#include<cstdio>
#include<queue>
#include<cstring>
#include<algorithm>
#define ll long long
using namespace std;
const int maxn=5e4+10;
int n;
ll ans;
struct fy
{
	ll x;
	bool operator<(const fy&a)
	const{return x>a.x;};
};
struct ffy
{
	ll x;
	bool operator<(const ffy&a)
	const{return x<a.x;};
};
priority_queue<fy>q;
priority_queue<ffy>qq;
int main()
{
	scanf("%d",&n);ll a,b;
	for(int i=1;i<=n;i++)
	{
		scanf("%lld",&a);
		q.push((fy){a});
		qq.push((ffy){a});
	}
	scanf("%lld",&a);
	ll ma,mi;
	while(!q.empty())
	{
		ma=q.top().x;q.pop();
		if(q.empty())break;
		b=q.top().x;q.pop(); 
		q.push((fy){ma*b+1});
	}
	while(!qq.empty())
	{
		mi=qq.top().x;qq.pop();
		if(qq.empty())break;
		b=qq.top().x;qq.pop();
		qq.push((ffy){mi*b+1});
	}
	ans=ma-mi;
	printf("%lld\n",ans);
	return 0;
}
```
