# 「一本通 1.1 例 2」种树

`一本通` `贪心`

- 区间选点问题

```cpp
#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
const int maxn=3e4+10;
int n,m;
bool use[maxn]; 
struct fy{int x,y,d;bool operator<(const fy&a)const{return y<a.y;};}q[maxn];
int main()
{
	scanf("%d%d",&n,&m);
	for(int i=1;i<=m;i++)scanf("%d%d%d",&q[i].x,&q[i].y,&q[i].d);
	sort(q+1,q+1+m);
	for(int i=1;i<=m;i++)
	{
		for(int j=q[i].x;j<=q[i].y;j++)
		{
			if(use[j])q[i].d--;
			if(!q[i].d)break;
		}
		if(q[i].d>0)for(int j=q[i].y;j>=q[i].x;j--)
		{
			if(!use[j])
			{
				use[j]=true;
				q[i].d--;
				if(!q[i].d)break;
			}
		}
//		for(int j=1;j<=n;j++)printf("%d ",use[j]);
//		printf("\n");
	}
	int ans=0;
	for(int i=1;i<=n;i++)if(use[i])ans++;
	printf("%d\n",ans);
	return 0;
}
```
