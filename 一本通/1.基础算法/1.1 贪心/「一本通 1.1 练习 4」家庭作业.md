# 「一本通 1.1 练习 4」家庭作业

`一本通` `贪心`

- 带限期和罚款的单位时间任务调度
- 单调队列

```cpp
#include<cstdio>
#include<queue>
#include<cstring>
#include<algorithm>
using namespace std;
const int maxn=1e6+10;
int n,l;
struct fy
{
	int x,y;
	bool operator<(const fy&a)
	const{return x<a.x;};
}q[maxn];
struct ffy
{
	int x;
	bool operator<(const ffy&a)
	const{return x>a.x;}
};
priority_queue<ffy>qq;
int main()
{
	scanf("%d",&n);
	for(int i=1;i<=n;i++)scanf("%d%d",&q[i].x,&q[i].y);
	sort(q+1,q+1+n);
	for(int i=1;i<=n;i++)
	{
		int a=q[i].x;
		int b=q[i].y;
		if(l<a)
		{
			qq.push((ffy){b});
			l++;
		}
		else 
		{
			int c=qq.top().x;
			if(b>c)
			{
				qq.pop();
				qq.push((ffy){b});
			}
		}
	}
	int ans=0;
	while(!qq.empty())
	{
		ans+=qq.top().x;
		qq.pop();
	}
	printf("%d\n",ans);
	return 0;
}
```
